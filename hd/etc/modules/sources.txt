%( op_m=1 Sources  %)
%( op_m=2 Sources complete (sources de membres de la famille) %)

%if;(has_linked_page.BIBLIO and linked_page.BIBLIO!="")
  <h2>[*bibliography]</h2>
  <ul>
    <li>%apply;capitalize(linked_page.BIBLIO)</li>
  </ul>
%end;
%reset_count;
%if;has_sources;%incr_count;%end;
%foreach;event;
  %if;event.has_src;%incr_count;%end;
%end;
%if;(count>0)
  %if;(count=1)
    <h2>[*source/sources]0</h2>
  %else;
    <h2>[*source/sources]1</h2>
  %end;
  <ul>
  %foreach;source;
    %if;(event_type="person" or
         event_type="family"
        )%nn;
      <li>%apply;capitalize(source_type)[:] %source;.</li>
    %end;
  %end;
  %reset_count;
  %empty_sorted_list;
  %foreach;event;
    %if;event.has_src;
      %if;(
        event.name!="special" and
        event.name!="special1" and
        event.name!="special2")
        %apply;add_in_sorted_list%with;
          %if;(event.date.year!="")%event.date.year;%else;9999%end;%and;
          %event.date.month;%and;
          %event.date.day;%and;
          <li>%apply;capitalize(event.name)%nn;
          %if;(event.type="fevt" and has_families and nb_families>1)
            %sp;([with] %event.spouse;)%nn;
          %end;[:] %event.src;.</li>
        %end;
      %end;
      %if;((event.name="special" or
           event.name="special1" or
           event.name="special2") and count=0)
        %apply;add_in_sorted_list%with;
          "9999"%and;
          ""%and;
          ""%and;
          <li>[timeline][:] %linked_page.TIMELINE;
        %end;
        %incr_count;
      %end;
    %end;
  %end;
  ---
  %foreach;sorted_list_item;
    %item.4;
  %end;
  </ul>
%end;

%reset_count;
%if;has_parents;
  %if;(father.has_sources or mother.has_fsources)%incr_count;%end;
%end;
%if;(has_siblings)
  %foreach;father.child;
    %if;((not is_self) and child.has_sources)%incr_count;%end;
  %end;
%end;
%if;(count>0 and op_m=2)
  <h5>[*other sources potentially usefull]</h5>
  <ul>
  %if;has_parents;
    <li>[*source/sources]0 [marriage event] [parents][:] %father.marriage_source;</li>
    <li>[*source/sources]0 [family/families]0 [parents][:] %father.fsources;</li>
  %end;
%(
  %if;(has_siblings)
    %foreach;father.child;
      %if;((not is_self) and child.psources!="")%nn;
        <li>[*source/sources]0 [brother/sister/sibling]s (%child.first_name;)[:] %child.psources;</li>%end;
    %end;
  %end;
%)
  %if;has_families;
    %foreach;family;
      %if;has_children;
        %foreach;child;
          %if;has_families;
            %foreach;family;
              %if;(marriage_source!="")<li>[*source/sources]0 [marriage event] [child/children]0 (%first_name;)[:] %marriage_source;</li>%end;
              %if;has_fsources;<li>[*source/sources]0 [family/families]0 [child/children]0 (%first_name;)[:] %fsources;</li>%end;
            %end;
          %end;
        %end;
      %end;
    %end;
  %end;

  </ul>
%end;
